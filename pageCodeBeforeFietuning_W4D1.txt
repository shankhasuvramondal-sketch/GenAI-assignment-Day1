  /** Fill the Parent Account field. */

  /** Click the lookup icon next to Parent Account. */

    await this.firstName.fill(firstName);

  /** Fill the Last Name field. */

    await this.annualRevenue.fill(amount);

    await this.email.fill(email);

  /** Fill the primary web URL. */
  async setWebUrl(url: string): Promise<void> {
    await this.webUrl.fill(url);
  }

  /** Fill the "To Name" field of the primary address. */
  async setToName(name: string): Promise<void> {
    await this.toName.fill(name);

  /** Click the "Create Lead" submit button. */
  async submit(): Promise<void> {
    await this.submitButton.click();
  }

  /**
   * Convenience method to fill the minimal required fields and submit.
   * Adjust as needed for your test scenarios.
   */
  async createLead(data: {
    companyName: string;
    firstName: string;
    lastName: string;
    primaryEmail?: string;
  }): Promise<void> {
    await this.setCompanyName(data.companyName);
    await this.setFirstName(data.firstName);
    await this.setLastName(data.lastName);
    if (data.primaryEmail) {
      await this.setEmail(data.primaryEmail);
    }
    await this.submit();
  }
